456 subroutine block_interp_grid(mask,xF,numR,xpe,xgd,Tcoff_line)

468 weightindex=2.d0
469 dxmax^d=2.d0*dl;
470 ixo^l=ixm^ll;
471 do i=1,numfl

480 {if (ixcmin^d<ixomin^d) ixcmin^d=ixomin^d\}
481 {if (ixcmax^d>ixomax^d) ixcmax^d=ixomax^d\}
