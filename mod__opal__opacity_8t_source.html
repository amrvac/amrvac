58  do while (logkappa_out .gt. 9.0d0)

61  enddo
62

64  do while (logkappa_out .eq. 0.0d0)
65  logr_in = logr_in - 0.5d0
66  call get_kappa(kappa_vals, logr_list, logt_list, logr_in, logt_in, logkappa_out)
67  enddo
68
