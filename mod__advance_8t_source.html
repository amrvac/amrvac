18 use mod_global_parameters
19 use mod_particles, only: handle_particles
20 use mod_source, only: add_split_source

23
24 integer :: iigrid, igrid, idimsplit

26 ! split source addition

28

31 ! do the sweeps in order of increasing idim,
32 do idimsplit=1,ndim

668 !> Advance a single grid over one partial time step
669 subroutine advect1_grid(igrid,method,qdt,dtfactor,ixI^L,idim^LIM,qtC,sCT,qt,s,dxs,x)
670
671 ! integrate one grid by one partial step
672 use mod_finite_volume
673 use mod_finite_difference
674 use mod_tvd

682 integer, intent(in) :: ixi^l, idim^lim
